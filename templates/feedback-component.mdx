<script>
// Handle banner close button
document.addEventListener('click', function(e) {
  if (e.target.classList.contains('banner-close')) {
    e.target.closest('.feedback-banner').remove();
  }
});

// Handle form submission
document.addEventListener('DOMContentLoaded', function() {
  const submitButton = document.getElementById('submit-feedback');
  const feedbackForm = document.getElementById('feedback-form');
  
  if (submitButton && feedbackForm) {
    submitButton.addEventListener('click', function() {
      const category = document.getElementById('feedback-category').value;
      const message = document.getElementById('feedback-message').value;
      
      // In a real implementation, you would send this data to your backend
      console.log('Feedback submitted:', { category, message });
      
      // Show success message
      if (window.feedbackComponent) {
        window.feedbackComponent.showToast('Thank you for your detailed feedback!', 'success');
      }
      
      // Close modal
      const modal = document.getElementById('feedback-modal');
      if (modal && window.feedbackComponent) {
        window.feedbackComponent.closeModal(modal);
      }
      
      // Reset form
      feedbackForm.reset();
    });
  }
});
</script>
