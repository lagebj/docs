## Core Web Vitals

Core Web Vitals are a set of metrics that measure user experience on the web. They consist of three metrics:

### 1. Largest Contentful Paint (LCP)

Measures loading performance. To provide a good user experience, LCP should occur within 2.5 seconds of when the page first starts loading.

<div id="lcp-metric" class="metric-card">
  <div class="metric-value">--</div>
  <div class="metric-label">LCP (seconds)</div>
  <div class="metric-status">Not measured</div>
</div>

### 2. First Input Delay (FID)

Measures interactivity. To provide a good user experience, pages should have a FID of 100 milliseconds or less.

<div id="fid-metric" class="metric-card">
  <div class="metric-value">--</div>
  <div class="metric-label">FID (milliseconds)</div>
  <div class="metric-status">Not measured</div>
</div>

### 3. Cumulative Layout Shift (CLS)

Measures visual stability. To provide a good user experience, pages should maintain a CLS of 0.1 or less.

<script dangerouslySetInnerHTML={{
  __html: `
  // Simple performance monitoring script
  document.addEventListener('DOMContentLoaded', () => {
    // Measure page load metrics
    window.addEventListener('load', () => {
      // LCP simulation (in a real implementation, we'd use the web-vitals library)
      const lcpMetric = document.getElementById('lcp-metric');
      if (lcpMetric) {
        // Simulate LCP measurement
        const lcpValue = performance.getEntriesByType('largest-contentful-paint')[0]?.startTime || 
                         (performance.timing.loadEventEnd - performance.timing.navigationStart);
        const lcpSeconds = (lcpValue / 1000).toFixed(2);
        
        lcpMetric.querySelector('.metric-value').textContent = lcpSeconds;
        lcpMetric.querySelector('.metric-status').textContent = 
          lcpValue < 2500 ? 'Good' : lcpValue < 4000 ? 'Needs improvement' : 'Poor';
        lcpMetric.querySelector('.metric-status').className = 'metric-status ' + 
          (lcpValue < 2500 ? 'good' : lcpValue < 4000 ? 'needs-improvement' : 'poor');
      }
      
      // FID simulation
      const fidMetric = document.getElementById('fid-metric');
      if (fidMetric) {
        // Simulate FID measurement (would normally be measured on first interaction)
        const fidValue = Math.random() * 200; // Random value for demo purposes
        fidMetric.querySelector('.metric-value').textContent = fidValue.toFixed(0);
        fidMetric.querySelector('.metric-status').textContent = 
          fidValue < 100 ? 'Good' : fidValue < 300 ? 'Needs improvement' : 'Poor';
        fidMetric.querySelector('.metric-status').className = 'metric-status ' + 
          (fidValue < 100 ? 'good' : fidValue < 300 ? 'needs-improvement' : 'poor');
      }
      
      // CLS simulation
      const clsMetric = document.getElementById('cls-metric');
      if (clsMetric) {
        // Simulate CLS measurement (would normally be measured over time)
        const clsValue = Math.random() * 0.3; // Random value for demo purposes
        clsMetric.querySelector('.metric-value').textContent = clsValue.toFixed(3);
        clsMetric.querySelector('.metric-status').textContent = 
          clsValue < 0.1 ? 'Good' : clsValue < 0.25 ? 'Needs improvement' : 'Poor';
        clsMetric.querySelector('.metric-status').className = 'metric-status ' + 
          (clsValue < 0.1 ? 'good' : clsValue < 0.25 ? 'needs-improvement' : 'poor');
      }
    });
  });
`
}} />
